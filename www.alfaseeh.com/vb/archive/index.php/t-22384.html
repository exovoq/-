ابني فديتك حاذر أن تكـن عجـلا = ولا بطيئـاً ، وسوّافـا ، ومتّكـلا
فللرّجولـة سيمـاء ، ومحـتـدم = وقـد خبرتـك لمّاحـاً ومكتمـلا
لا تدّعي في الورى ما لست تملكه = فالفضل للمرء فيما ينجـز العمـل
صادق بصدق ، وإن فارقت لا ندم = وسامح الناس فيما يبعـث الأمـل
زر من تحب ، وعاشر بالّتي أحسن = وللمكارم ، والأرحام كن وصـلا
كن بالمروءات والإنصاف متّصفاً = وللجميل أسيـرا تُنتخـى رجـلا
وراقب الله فـي طيـب تعطّـره = أرومة أنت منها الطفـل والكهـل
ولا تفاخر بأصـل أنـت وارثـه = فما تحقّق يعطـي الفخـر والنبـل
إذا دعيت فلبـي تكتسـب فضـلا = وإن دعوت فأكثر مرحبـا وهـلا
تجنّب المنّ والإسـراف والبخـل = ولاطف الناس لا ضعفا ولا كسلا
وكن لجـارك رفـداً خيّـراً أبـداً = فحقّـه ثابـت لا يقبـل المـطـل
وإن بدت لك منه هفـوة فاصفـح = وبادر السوء بالإحسـان متصـلا
واختر لنفسك منهاجـاً ومتّصفـاً = فللرجال صفـات ترتجـى مثـلا
رافق قريناً حصيفاً نفحـه أرجـاً = يهديك من نفحه إن غاب أو وصلا
ولا تـؤمّ مكانـاً أنـت ترهـبـه = كي لا يكـون لا فّـاق عليـك دلا
أحذر صديقاً بأعمال العدى اشتغل = فمن تهاون جـاء النـاس مبتـذلا
وراقب الله في سـرّ وفـي علـن = وكل أمر له فـي النـاس معتمـل

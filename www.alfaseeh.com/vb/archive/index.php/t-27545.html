لا تسجدي


لا تنحني فتقبلي منِّي يــــدي = وأمام أقدامي فلا , لا تسـجدي
أنا لست أرضى بالهوان من الهوى = فتكبري وتجبري وتشددي
ولتخلعي هيا رداء مذلـــــــــــة = ورداء عز وافتخار فارتـــدي
ولتعلني حرباً علىّ شــــديدة = وسهامَ عينيكِ بقلبي أغمـــدي
كوني جليداً، زمهريرًا، عاصــفاً = كوني حريقاً, ألهبيني, بـددي
كوني سماء ولتضِني بالحيـــا = إن كنت أرجوه , وإلا فاعتـــدي
كوني خريفاً , دمريني, حطمي = وبحِيرتي وبشِقوتي فلتســـعدي
وابكي إذا رمت ابتسامك مــرة = وإنِ اشتهيت دموع عينك فاجمدي
كوني ملاكاً إن أردتك فاجــرة = وإذا رجوتك برة فتمــــردي
كوني لهيبا إن أتيت مفارقـــاً = وإذا نشدت محبة, فتبلـــدي
كوني عقاباً لو أردت ســـلامة = ولو ابتغيت الهجر منك تــوددي
وإذا أردت الصفح عني فاشــــددي = ولئن أُرِدْ جفوا فقولي: سـيدي
ولئن تمنيت الوعيد فواعـــدي = وإذا رجوت الوعد فلتتوعـدي
فلكم أمل من الرتابة فى الهـــوى = فتصنعي وتمنعي وتجــددي
فالماء يأسَن عذبه إن لم يســـر = وإذا جرى, أضحى شهيَّ المورد
والبدر إن لم يستتر بمــــداره = ما كان أشبه حاله بالفرقــــد
هيا لنصنع فى الغرام حكايـــة = مشبوبة وبها الأحبة تقتــــدي
فالحب نار والجليد وقــــودها = وهو الورود على جبيل أجــرد
وصل، وهجـــر ، عزة، وتذلل = غدر ، وفا ، أمن، وخوف من غد
طهر، ورجس ،غيرة، وتبلـــد = يأس، رجاء ، كل ذلك فاشــهدي
لكم اشتكى قلبي اعتداء لحاظـها= وأنا أمام العالمين المعتــــدي
ويروقني شجن وســــهد دائم = ويطيب لي أن تعمدي فلتعمدي
أنا لست أنسى حبها مهما جـرى = ولئن جرى , فلمن تكون قصائدي؟
فبغير حبي لا تديني مذهبـــا = ولغير قلبي في الهوى لا تسـجدي

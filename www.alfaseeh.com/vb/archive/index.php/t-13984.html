بين قبيلتين شهيرتين نشبت حربٌ هجائية شعرية ، أخذتا تتلاطما بالشعر عبر آلة البلوتوث ، بعد أن نشب بينهما عراك، بين بعض أفرادهما ، سالت فيها الدماء فعلا وكانت المشكلة تتفاقم لولا توفيق الله . وبما أني أحد أبناء القبيلتين المتخاصمتين جاشت نفسي بهذه الأبيات .

دَعوا عنكم لهيب الحرب إنّ الـــ
منايا الحمر تُلفي بالدمــــــــاءِ

وصونوا عن مساعيكم بتقوى الـ
إلـه الحق خيرٌ من رغـــــــــاءِ

ولا تلقــوا بأيديـــــــــــكم بــــوادٍ
يتيهُ الحــرُّ فيه بالعـــــــــــــراء

ومن هَوِيَ الخصـــامَ بفعلِ غــــرِّ
يلاقي الهول من ضرم الشـــواء

فهل تُجنى ثمـــــــــــارُ البغي إلا
حصيداً من قتيلٍ أو تنـــــــــــــاءِ

ألا قــــومُ تآخـــــــــوا واستناروا
يُردُّ الغدرُ جزلاً بالوفـــــــــــــاءِ

فهذي سنة الهــــــــادي فصيروا
تِباعَ الهدي ما عقبى العَــــــداءِ ؟

لنا الإســـــــــــلام شرع مستقيم
يفضُّ النزع لُقياً بالإخـــــــــــاءِ

وما ركْب الأعــــادي حين ألفوا
علينا إلا من نكس الـلـــــــــــواء

ألا حلفٌ تحالف ضــــــد حلفٍ
يسوم العُرْبَ مكراً باعتــــــــداءِ

ولاحت نـــــــــــارعاديةٍ فتغدو
سهامُ الويل تحكم بافتــــــــــــراءِ

وصار الحق مأسوراً بحلف الـ
أعادي يُرتجى منهم شقائــــــــــي

فهبوا من سباتــكمُ وشيــــــدوا
لنا الأمجاد نسمو للإبــــــــــــاء

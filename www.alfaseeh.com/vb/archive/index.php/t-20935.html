لمـن طلـل بيـن الجديـة والجـبـلِ = مـحل قديـم العهـد طالـت بـه الطيـلِ
عفـا غيـر مرتـاد ومـر كسـرحـب = ومنخفـض طـام تنـكـر واضمـحـل
وزالت صروف الدهر عنـه فأصبحـت=على غير سكـان ومـن سكـن ارتحـل
تنـطـح بـالاطـلال مـنـه مجلـجـل = أحـم إذا احمومـت سحائبـه انسـجـل
بريـح وبــرق لاح بـيـن سحـائـب = ورعـد إذا مـا هـب هاتـفـه هـطـل
فانبـت فيـه مـن غشنـض وغشنـض = ورونـق رنـد والصلـنـدد والأســل
وفيـه القطـا والبـوم وابـن حبـوكـل = وطيـر القطـاط والبلـنـدد والحـجـل
وعنـثـلـة والخـيـثـوان وبـرســل = وفـرخ فـريـق والرفـلـة والـرفـل
وفـيـل واذيــاب وابــن خـويـدر = وغنسلـة فيهـا الخفيعـان قـد نــزل
وهــام وهمـهـام وطـالـع انـجــد = ومنحبـك الروقيـن فـي سيـره مـيـل
فلمـا عرفـت الـدار بـعـد توهـمـي = تكفكـف دمعـي فـوق خـدي وانهمـل
فقلـت لهـا يـادار سلمـى ومـا الـذي = تمتـعـت لا بـدلـت يــادار بالـبـدل
لقد طـال مـا اضحيـت قفـرا ومالفـا = ومنتظـرا للحـي مـن حـل او رحـل

لقد كنت اسبـي الغيـد أمـرد نـا شئـا = ويسبيننـي منـهـن بـالـدل والمـقـل
ليـالـي أسـبـي الغانـيـات بجـمـة = معثكـلـة ســوداء زينـهـا رجــل
كـأن قطيـر الـبـان فــي عنكاتـهـا = علـى منثنـى والمنكبيـن عطـى رطـل

لأصـبـح مفتـونـا معـنـى بحبـهـا = كـأن لـم يصـم لله يومـا ولـم يصـل
ألا رب يــوم قــد لـهـوت بدلـهـا = إذا مـا أبوهـا ليلـة غـاب أو غـفـل
فقالـت لأتـراب لـهـا قــد رميـتـه = فكيـف بـه إن مـات أو كيـف يحتبـل
